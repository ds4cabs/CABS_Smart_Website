### Step 2: Import Necessary Libraries
```python
from Bio import SeqIO
import pandas as pd
import numpy as np
```

### Step 3: Load DNA Sequences
```python
def load_sequences(file_path):
    sequences = []
    for record in SeqIO.parse(file_path, "fasta"):
        sequences.append(str(record.seq))
    return sequences

dna_sequences = load_sequences("genes.fasta")
```

### Step 4: Define a Function to Find Motifs
```python
def find_motifs(sequences, motif):
    motif_locations = {}
    for seq in sequences:
        locations = [i for i in range(len(seq)) if seq.startswith(motif, i)]
        motif_locations[seq] = locations
    return motif_locations

motif = "ATG"  # Example motif for start codon
motif_locations = find_motifs(dna_sequences, motif)
```

### Step 5: Analyze Motif Distribution
```python
def analyze_motifs(motif_locations):
    data = []
    for seq, locations in motif_locations.items():
        data.append({'Sequence': seq, 'MotifCount': len(locations), 'Locations': locations})
    
    df = pd.DataFrame(data)
    return df

motif_analysis_df = analyze_motifs(motif_locations)
print(motif_analysis_df)
```

### Step 6: Visualize Motif Distribution (Optional)
```python
import matplotlib.pyplot as plt

plt.bar(motif_analysis_df['Sequence'], motif_analysis_df['MotifCount'])
plt.xlabel('DNA Sequences')
plt.ylabel('Number of Motifs Found')
plt.title('Motif Distribution in DNA Sequences')
plt.xticks(rotation=90)
plt.tight_layout()
plt.show()
```
