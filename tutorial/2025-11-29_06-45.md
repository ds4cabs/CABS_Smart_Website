```python
import numpy as np

from torch.utils.data import DataLoader, Dataset
```

### Step 3: Data Preparation
Assume we have a FASTA file of protein sequences. We will convert these sequences into a numerical format suitable for input into a neural network.

```python
class ProteinDataset(Dataset):
    def __init__(self, fasta_file):
        self.sequences = []
        self.labels = []
        
        for record in SeqIO.parse(fasta_file, "fasta"):
            self.sequences.append(self.encode_sequence(str(record.seq)))
            self.labels.append(self.encode_label(record.description.split()[1]))  # Example for extracting label

    def encode_sequence(self, seq):
        amino_acids = "ACDEFGHIKLMNPQRSTVWY"  # Standard amino acids
        mapping = {aa: idx for idx, aa in enumerate(amino_acids)}
        return [mapping[aa] for aa in seq if aa in mapping]

    def encode_label(self, label):
        le = LabelEncoder()
        return le.fit_transform([label])
        
    def __len__(self):
        return len(self.sequences)

    def __getitem__(self, idx):
        return torch.tensor(self.sequences[idx]), torch.tensor(self.labels[idx])
```

### Step 4: Create the Neural Network
```python
class ProteinNN(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(ProteinNN, self).__init__()
        self.lstm = nn.LSTM(input_size, hidden_size, batch_first=True)
        self.fc = nn.Linear(hidden_size, output_size)

    def forward(self, x):
        out, _ = self.lstm(x)
        return self.fc(out[:, -1, :])
```

### Step 5: Training the Model
```python
def train_model(dataloader, model, criterion, optimizer, epochs):
    for epoch in range(epochs):
        for sequences, labels in dataloader:
            optimizer.zero_grad()
            output = model(sequences.float())
            loss = criterion(output, labels.float())
            loss.backward()
            optimizer.step()
    print(f'Epoch {epoch + 1}, Loss: {loss.item()}')
```

### Step 6: Putting It All Together
```python
# Constants and Hyperparameters
INPUT_SIZE = 20  # Number of amino acids
HIDDEN_SIZE = 128
OUTPUT_SIZE = 3  # Example for 3 types of secondary structure
EPOCHS = 10
BATCH_SIZE = 16

# Load Data
dataset = ProteinDataset('proteins.fasta')
dataloader = DataLoader(dataset, batch_size=BATCH_SIZE, shuffle=True)

# Initialize Model, Loss, and Optimizer
model = ProteinNN(INPUT_SIZE, HIDDEN_SIZE, OUTPUT_SIZE)
criterion = nn.MSELoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# Train the Model
train_model(dataloader, model, criterion, optimizer, EPOCHS)
```
