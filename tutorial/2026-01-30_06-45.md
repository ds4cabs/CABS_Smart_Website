### Step 1: Data Preparation

First, we will load a dataset of protein sequences and their corresponding secondary structures. Here, we will use the CB513 dataset as an example.

```python
import pandas as pd
from Bio import SeqIO

# Load protein sequences and secondary structures
def load_data(file_path):
    sequences, structures = [], []
    for record in SeqIO.parse(file_path, "fasta"):
        sequences.append(str(record.seq))
        structures.append(get_secondary_structure(record.id))
    return sequences, structures

def get_secondary_structure(protein_id):
    # This function should return the secondary structure for a given protein ID
    # For simplicity, we are mapping ids for an illustrative example
    structure_map = {
        "1A1X": "H",
        "1A2X": "E",
        # Add all mappings as needed
    }
    return structure_map.get(protein_id, "C")  # Default to coil

sequences, structures = load_data("cb513.fasta")
```

### Step 2: Encoding Sequences

We need to convert the amino acid sequences into a numerical format suitable for neural networks.

```python
import numpy as np

# Encode sequences to integers
def encode_sequences(sequences):
    amino_acids = 'ACDEFGHIKLMNPQRSTVWY'  # Standard amino acids
    amino_acid_dict = {amino_acid: index for index, amino_acid in enumerate(amino_acids)}
    
    encoded = []
    for seq in sequences:
        encoded_seq = [amino_acid_dict[aa] for aa in seq if aa in amino_acid_dict]
        encoded.append(encoded_seq)
    
    return np.array(encoded)

X = encode_sequences(sequences)
```

### Step 3: Preparing Labels

Convert the secondary structures into one-hot encoded format.

```python
from sklearn.preprocessing import OneHotEncoder

# One-hot encode secondary structures
encoder = OneHotEncoder(sparse=False)
Y = encoder.fit_transform(np.array(structures).reshape(-1, 1))
```

### Step 4: Building the Neural Network

Create a simple feedforward neural network using TensorFlow.

```python
import tensorflow as tf

# Define model
model = tf.keras.Sequential([
    tf.keras.layers.Input(shape=(None,)),
    tf.keras.layers.Embedding(input_dim=len('ACDEFGHIKLMNPQRSTVWY'), output_dim=128),
    tf.keras.layers.LSTM(128, return_sequences=True),
    tf.keras.layers.Dense(len(encoder.categories_[0]), activation='softmax')
])

model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
model.summary()
```

### Step 5: Training the Model

Train the model on the protein sequences and their corresponding secondary structures.

```python
# Pad sequences to the same length for training
from keras.preprocessing.sequence import pad_sequences

X_padded = pad_sequences(X, padding='post')

# Train the model
model.fit(X_padded, Y, epochs=50, batch_size=32, validation_split=0.2)
```
