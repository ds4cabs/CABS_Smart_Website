### 2. Import Libraries

```python
import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Conv2D, MaxPooling2D, Flatten, Dense, Dropout
from Bio import SeqIO
```

### 3. Prepare Dataset
Assume we have a FASTA file containing protein sequences. We will create feature vectors for training. 

```python
def load_sequences(filename):
    sequences = []
    for record in SeqIO.parse(filename, "fasta"):
        sequences.append(str(record.seq))
    return sequences

sequences = load_sequences("protein_sequences.fasta")
```

### 4. Encode Sequences to Vectors
For this example, we will use a simple one-hot encoding for the amino acids.

```python
def one_hot_encode(sequences):
    amino_acids = 'ACDEFGHIKLMNPQRSTVWY'
    amino_dict = {aa: idx for idx, aa in enumerate(amino_acids)}
    encoded_seqs = []

    for seq in sequences:
        one_hot = np.zeros((len(seq), len(amino_acids)), dtype=int)
        for i, aa in enumerate(seq):
            if aa in amino_dict:
                one_hot[i][amino_dict[aa]] = 1
        encoded_seqs.append(one_hot)
    
    return np.array(encoded_seqs)

encoded_sequences = one_hot_encode(sequences)
```

### 5. Build CNN Model

```python
def build_model(input_shape):
    model = Sequential()
    model.add(Conv2D(32, (3, 3), activation='relu', input_shape=input_shape))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Conv2D(64, (3, 3), activation='relu'))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Flatten())
    model.add(Dense(128, activation='relu'))
    model.add(Dropout(0.5))
    model.add(Dense(1, activation='sigmoid'))  # Binary classification for simplicity
    model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])
    return model

input_shape = (encoded_sequences.shape[1], encoded_sequences.shape[2], 1)  # Adjust based on input
model = build_model(input_shape)
```

### 6. Train the Model

```python
# Assuming we have labels for the sequences (for example: 0 or 1)
labels = np.random.randint(0, 2, size=(len(encoded_sequences), 1))  # Dummy labels for demonstration

# Reshape for the model input
X_train = encoded_sequences.reshape(encoded_sequences.shape[0], encoded_sequences.shape[1], encoded_sequences.shape[2], 1)

model.fit(X_train, labels, epochs=10, batch_size=32, validation_split=0.2)
```

### 7. Evaluate the Model

```python
loss, accuracy = model.evaluate(X_train, labels)
print(f'Model Loss: {loss}, Model Accuracy: {accuracy}')
```

Utilizing convolutional neural networks in Python allows for effective analysis of protein sequences in bioinformatics, enabling advanced prediction capabilities. The approach above demonstrates a foundational method to start from sequence data to building predictive models.
