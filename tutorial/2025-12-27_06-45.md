### Step 2: Import Libraries

```python
import numpy as np
import pandas as pd
from Bio import SeqIO
from sklearn.model_selection import train_test_split
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, LSTM, Embedding
from tensorflow.keras.utils import to_categorical
```

### Step 3: Data Preparation

We'll use the CB513 dataset, which contains protein sequences and their corresponding secondary structure annotations.

```python
def load_data(file_path):
    sequences = []
    labels = []
    
    for record in SeqIO.parse(file_path, "fasta"):
        sequence = str(record.seq)
        secondary_structure = record.description.split()[1]  # Assuming structure in description
        sequences.append(sequence)
        labels.append(secondary_structure)
        
    return sequences, labels

file_path = 'cb513.fasta'  # Update this to the path of your dataset
sequences, labels = load_data(file_path)
```

### Step 4: Encode Sequences and Labels

```python
def encode_sequences(sequences):
    amino_acids = 'ACDEFGHIKLMNPQRSTVWY'
    encoding_dict = {aa: i + 1 for i, aa in enumerate(amino_acids)}  # Start from 1
    encoded_sequences = [[encoding_dict[aa] for aa in seq] for seq in sequences]
    return np.array(encoded_sequences)

def encode_labels(labels):
    label_dict = {'H': 0, 'E': 1, 'C': 2}  # H: Helix, E: Strand, C: Coil
    encoded_labels = [label_dict[label] for label in labels]
    return to_categorical(np.array(encoded_labels))

encoded_sequences = encode_sequences(sequences)
encoded_labels = encode_labels(labels)
```

### Step 5: Split the Dataset

```python
X_train, X_test, y_train, y_test = train_test_split(encoded_sequences, encoded_labels, test_size=0.2, random_state=42)
```

### Step 6: Build the Model

```python
model = Sequential()
model.add(Embedding(input_dim=21, output_dim=50, input_length=X_train.shape[1]))  # 20 amino acids + 1 for padding
model.add(LSTM(100, return_sequences=True))
model.add(Dense(3, activation='softmax'))  # Output layer for 3 classes

model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])
```

### Step 7: Train the Model

```python
model.fit(X_train, y_train, epochs=10, batch_size=64, validation_data=(X_test, y_test))
```

### Step 8: Evaluate the Model

```python
loss, accuracy = model.evaluate(X_test, y_test)
print(f"Test Accuracy: {accuracy * 100:.2f}%")
```
